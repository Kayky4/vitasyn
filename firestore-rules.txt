
rules_version = '2';

service cloud.firestore {
  match /databases/{database}/documents {

    // Funções auxiliares
    function isAuthenticated() {
      return request.auth != null;
    }

    function isOwner(userId) {
      return request.auth.uid == userId;
    }

    // Coleção de Usuários (Base)
    match /users/{userId} {
      allow read: if isAuthenticated() && isOwner(userId);
      allow write: if isAuthenticated() && isOwner(userId);
    }

    // Coleção de Profissionais (Perfil Público e Detalhado)
    match /professionals/{professionalId} {
      allow read: if true;
      allow create: if isAuthenticated() && isOwner(professionalId);
      allow update: if isAuthenticated() && isOwner(professionalId);
    }

    // Coleção de Consultas
    match /consultations/{consultationId} {
      // Leitura
      allow read: if isAuthenticated() && (
        resource.data.patientId == request.auth.uid || 
        resource.data.professionalId == request.auth.uid
      );

      // Criação
      allow create: if isAuthenticated() && (
        request.resource.data.patientId == request.auth.uid ||
        request.resource.data.professionalId == request.auth.uid
      );

      // Atualização
      allow update: if isAuthenticated() && (
        resource.data.professionalId == request.auth.uid ||
        resource.data.patientId == request.auth.uid
      );
      
      // Deletar: Permitido se o usuário for o profissional dono do agendamento
      allow delete: if isAuthenticated() && (
        resource.data.professionalId == request.auth.uid
      );
    }

    // Coleção de Transações
    match /transactions/{transactionId} {
        allow read: if isAuthenticated() && (
            resource.data.relatedId == request.auth.uid
        );
        allow write: if false; 
    }

    // Notificações
    match /notifications/{notifId} {
        allow read: if isAuthenticated() && resource.data.userId == request.auth.uid;
        allow update: if isAuthenticated() && resource.data.userId == request.auth.uid;
        allow create: if false;
    }
  }
}
